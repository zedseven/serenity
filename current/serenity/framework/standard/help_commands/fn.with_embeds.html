This is supported on crate features framework and standard_framework and cache and http only.
Expand description

Posts an embed showing each individual command group and its commands.

Examples

Use the command with StandardFramework::help:

use std::{collections::HashSet, hash::BuildHasher};

use serenity::{
    framework::standard::{
        help_commands::*,
        macros::help,
        Args,
        CommandGroup,
        CommandResult,
        HelpOptions,
        StandardFramework,
    },
    model::prelude::*,
};

#[help]
async fn my_help(
    context: &Context,
    msg: &Message,
    args: Args,
    help_options: &'static HelpOptions,
    groups: &[&'static CommandGroup],
    owners: HashSet<UserId>,
) -> CommandResult {
    let _ = with_embeds(context, msg, args, &help_options, groups, owners).await;
    Ok(())
}

let framework = StandardFramework::new().help(&MY_HELP);
